  const tipText = {

    1: "The administration has made election denialism official federal policy by using investigative and enforcement powers to undermine confidence in elections.",

    2: "The administration has used investigative, prosecutorial, and regulatory power to deter opposition, voter engagement, and faithful public service.",

    3: "The administration has used and threatened violence to quash protests and create conditions for voter intimidation.",

    4: "The administration is pursuing a coordinated effort to seize control over election administration and make it harder for millions of eligible Americans to vote.",

    5: "If the administration can't hold on to power by tilting the electoral playing field or making it harder for Americans to vote, they will contest, delay, or overturn results they don't like."

  };


  const tt = document.getElementById('tt');

  const wrap = document.getElementById('diagramWrap');

  const svg  = document.getElementById('mainSvg');


  // Convert SVG viewBox coords to percentage-based positions on the rendered element

  function svgToPercent(svgX, svgY, svgW, svgH) {

    return { left: (svgX / 600) * 100, top: (svgY / 800) * 100, w: (svgW / 600) * 100, h: (svgH / 800) * 100 };

  }


  // Hover zone definitions in SVG coords [x, y, w, h]

  const zones = {

    hz1: [215, 82,  170, 85],

    hz2: [205, 197, 190, 82],

    hz3: [100, 385, 150, 85],

    hz4: [345, 385, 150, 85],

    hz5: [235, 655, 130, 90],

  };


  function positionZones() {

    for (const [id, [x, y, w, h]] of Object.entries(zones)) {

      const p = svgToPercent(x, y, w, h);

      const el = document.getElementById(id);

      el.style.left   = p.left + '%';

      el.style.top    = p.top  + '%';

      el.style.width  = p.w    + '%';

      el.style.height = p.h    + '%';

    }

  }


  positionZones();

  window.addEventListener('resize', positionZones);


  // Tooltip behaviour

  document.querySelectorAll('.hz').forEach(hz => {

    hz.addEventListener('mouseenter', e => {

      tt.textContent = tipText[hz.dataset.tip];

      tt.classList.add('show');

    });

    hz.addEventListener('mousemove', e => {

      const r = wrap.getBoundingClientRect();

      let x = e.clientX - r.left + 14;

      let y = e.clientY - r.top  - 10;

      if (x + 214 > r.width) x = e.clientX - r.left - 224;

      if (y + tt.offsetHeight > r.height) y = e.clientY - r.top - tt.offsetHeight - 6;

      tt.style.left = x + 'px';

      tt.style.top  = y + 'px';

    });

    hz.addEventListener('mouseleave', () => tt.classList.remove('show'));

  });


  // Animation

  function fadeIn(el, duration) {

    return new Promise(resolve => {

      el.style.transition = `opacity ${duration}ms ease`;

      el.style.opacity = 1;

      setTimeout(resolve, duration);

    });

  }

  function resetEl(el) { el.style.transition = 'none'; el.style.opacity = 0; }

  const pause = ms => new Promise(r => setTimeout(r, ms));


  async function runAnimation() {

    const ids = ['anim-step1','anim-ring-base','anim-step2','anim-step3','anim-step4','anim-center-text','anim-step5'];

    ids.forEach(id => resetEl(document.getElementById(id)));

    const btn = document.getElementById('replayBtn');

    btn.disabled = true;


    await pause(300);

    await fadeIn(document.getElementById('anim-step1'), 900);

    await pause(600);

    await fadeIn(document.getElementById('anim-ring-base'), 600);

    await pause(100);

    await fadeIn(document.getElementById('anim-step2'), 300);

    await pause(1500);

    await fadeIn(document.getElementById('anim-step3'), 300);

    await pause(1500);

    await fadeIn(document.getElementById('anim-step4'), 300);

    await pause(1500);

    await fadeIn(document.getElementById('anim-center-text'), 700);

    await pause(1900);

    await fadeIn(document.getElementById('anim-step5'), 900);


    btn.disabled = false;

  }


  window.addEventListener('load', () => setTimeout(runAnimation, 400));

</script>
